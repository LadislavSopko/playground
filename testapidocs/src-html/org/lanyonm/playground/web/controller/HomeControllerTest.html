001package org.lanyonm.playground.web.controller;
002
003import static org.springframework.restdocs.mockmvc.MockMvcRestDocumentation.document;
004import static org.springframework.restdocs.mockmvc.MockMvcRestDocumentation.documentationConfiguration;
005import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.get;
006import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.post;
007import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.model;
008import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
009
010import org.junit.Before;
011import org.junit.Rule;
012import org.junit.Test;
013import org.junit.runner.RunWith;
014import org.lanyonm.playground.config.DataConfig;
015import org.lanyonm.playground.config.ViewResolver;
016import org.lanyonm.playground.config.WebConfig;
017import org.lanyonm.playground.service.ExceptionServiceImpl;
018import org.lanyonm.playground.service.TodoServiceImpl;
019import org.lanyonm.playground.service.UserServiceImpl;
020import org.springframework.beans.factory.annotation.Autowired;
021import org.springframework.mock.web.MockHttpSession;
022import org.springframework.restdocs.RestDocumentation;
023import org.springframework.test.context.ContextConfiguration;
024import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
025import org.springframework.test.context.web.AnnotationConfigWebContextLoader;
026import org.springframework.test.context.web.WebAppConfiguration;
027import org.springframework.test.web.servlet.MockMvc;
028import org.springframework.test.web.servlet.setup.MockMvcBuilders;
029import org.springframework.web.context.WebApplicationContext;
030
031/**
032 * @author lanyonm
033 */
034@RunWith(SpringJUnit4ClassRunner.class)
035@WebAppConfiguration
036@ContextConfiguration(loader=AnnotationConfigWebContextLoader.class, classes={DataConfig.class, ExceptionServiceImpl.class, TodoServiceImpl.class, UserServiceImpl.class, ViewResolver.class, WebConfig.class})
037public class HomeControllerTest {
038
039        @Rule
040        public final RestDocumentation restDocumentation = new RestDocumentation("target/generated-snippets");
041
042        @Autowired
043        private WebApplicationContext wac;
044
045        private MockMvc mockMvc;
046
047        @Before
048        public void setup() {
049                this.mockMvc = MockMvcBuilders.webAppContextSetup(this.wac)
050                                .apply(documentationConfiguration(this.restDocumentation))
051                                .build();
052        }
053
054        @Test
055        public void testIndex() throws Exception {
056                this.mockMvc.perform(get("/")).andExpect(status().isOk())
057                                .andExpect(model().attribute("example", "Hello World!"))
058                                .andDo(document("index"));
059        }
060
061        @Test
062        public void testUsers() throws Exception {
063                this.mockMvc.perform(get("/users")).andExpect(status().isOk())
064                                .andExpect(model().attributeExists("users"))
065                                .andDo(document("users"));
066        }
067
068        @Test
069        public void testExceptions() throws Exception {
070                this.mockMvc.perform(get("/exceptions")).andExpect(status().isOk())
071                                .andExpect(model().attribute("exceptionsState", false));
072        }
073
074        @Test
075        public void testToggleExceptions() throws Exception {
076                MockHttpSession session = new MockHttpSession();
077                session.setAttribute("exceptionsToggle", false);
078                this.mockMvc.perform(post("/exceptions").session(session)).andExpect(status().isOk())
079                                .andExpect(model().attribute("exceptionsState", true));
080                this.mockMvc.perform(post("/exceptions").session(session)).andExpect(status().isOk())
081                                .andExpect(model().attribute("exceptionsState", false));
082                this.mockMvc.perform(post("/exceptions").session(session)).andExpect(status().isOk())
083                                .andExpect(model().attribute("exceptionsState", true));
084                this.mockMvc.perform(post("/exceptions").session(session)).andExpect(status().isOk())
085                                .andExpect(model().attribute("exceptionsState", false));
086        }
087}
