001package org.lanyonm.playground.service;
002
003import static org.junit.Assert.assertEquals;
004import static org.junit.Assert.assertNotNull;
005import static org.junit.Assert.assertTrue;
006
007import java.util.List;
008
009import org.junit.Test;
010import org.junit.runner.RunWith;
011import org.lanyonm.playground.config.DataConfig;
012import org.lanyonm.playground.domain.User;
013import org.springframework.beans.factory.annotation.Autowired;
014import org.springframework.test.context.ContextConfiguration;
015import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
016import org.springframework.test.context.support.AnnotationConfigContextLoader;
017
018@RunWith(SpringJUnit4ClassRunner.class)
019@ContextConfiguration(loader=AnnotationConfigContextLoader.class, classes={DataConfig.class, UserServiceImpl.class})
020public class UserServiceTest {
021
022        @Autowired
023        private UserService userService;
024
025        @Test
026        public void testGetAllUsers() {
027                List<User> users = userService.getAllUsers();
028                assertNotNull("users should not be null", users);
029        }
030
031        @Test
032        public void testSaveUser() {
033                assertEquals("should have 1 user in the db", 1, userService.getAllUsers().size());
034                User user = new User();
035                user.setFirstName("Toby");
036                user.setLastName("Cat");
037                boolean ret = userService.saveUser(user);
038                assertTrue("should have saved Toby", ret);
039                user = userService.getAllUsers().get(1);
040                assertEquals("number of users should now be 2", 2, userService.getAllUsers().size());
041                assertTrue("Toby should now have an id", user.getFirstName().equals("Toby") && user.getId() > 0);
042                user.setFirstName("Neko");
043                userService.saveUser(user);
044                assertEquals("number of users should still be 2", 2, userService.getAllUsers().size());
045                assertEquals("the second user should be Neko", "Neko", userService.getAllUsers().get(1).getFirstName());
046        }
047}
